Kaggle Watch uses dweet.io to receive messages from kaggle code. Insert following code block in your python code to push print() logs to dweet server. 
            
class Logger(object):
    CHANNEL_NAME = 'my-channel'

    def __init__(self):
        self.terminal = sys.stdout

    def write(self, message):
        self.terminal.write(message + '\n')
        payload = {'msg': message}
        quoted = urlencode(payload)
        if quoted != 'msg=%0A':
            thr = threading.Thread(target=self.send, args=(quoted,), kwargs={})
            thr.start()

    def flush(self):
        pass

    @staticmethod
    def send(msg):
        msg = 'https://dweet.io/dweet/for/' + Logger.CHANNEL_NAME + '?' + msg
        try:
            request.urlopen(msg).read()
        except Exception as e:
            sys.stdout.terminal.write(e)


sys.stdout = Logger()
